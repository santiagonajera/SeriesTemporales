# Calcular el coeficiente de variación (CV = std / mean) para cada tienda
coef_var = {}

for tienda in data.columns:
    media = data[tienda].mean()
    desviacion_estandar = data[tienda].std()
    coef_var[tienda] = desviacion_estandar / media

# Convertir el resultado a un DataFrame para mejor visualización
cv_df = pd.DataFrame.from_dict(coef_var, orient='index', columns=['Coeficiente de Variación'])

# Crear un DataFrame con los datos ajustados por winsorización
df_winsorized = pd.DataFrame({
    'Fecha': serie_tienda_1.index,
    'Winsorizada': winsorized_data
}).set_index('Fecha')

# Realizar la descomposición estacional (aditiva) de la serie winsorizada
descomposicion_winsor = seasonal_decompose(df_winsorized['Winsorizada'], model='additive', period=12)

# Extraer la tendencia y estacionalidad de la serie winsorizada
tendencia_winsor = descomposicion_winsor.trend.dropna()  # Eliminar NaN para evitar problemas
estacionalidad_winsor = descomposicion_winsor.seasonal[-12:]  # Últimos 12 meses de estacionalidad (1 ciclo completo)

# Proyectar la tendencia para los próximos 12 meses
# Ajustamos un modelo lineal a los datos de la tendencia existente para hacer la extrapolación
meses_futuros = 12
tendencia_futura = np.polyval(np.polyfit(range(len(tendencia_winsor)), tendencia_winsor, 1), 
                              range(len(tendencia_winsor), len(tendencia_winsor) + meses_futuros))

# Repetir la estacionalidad para los próximos 12 meses
estacionalidad_futura = np.tile(estacionalidad_winsor, int(np.ceil(meses_futuros / 12)))[:meses_futuros]

# Sumar la tendencia proyectada y la estacionalidad para generar el forecast
forecast_futuro = tendencia_futura + estacionalidad_futura

# Crear las fechas para los próximos 12 meses
fechas_futuras = pd.date_range(start=data.index[-1] + pd.DateOffset(months=1), periods=meses_futuros, freq='MS')

# Crear un DataFrame con las proyecciones
forecast_df = pd.DataFrame({
    'Fecha': fechas_futuras,
    'Forecast': forecast_futuro
}).set_index('Fecha')

# Concatenar los datos originales con los datos proyectados
df_completo = pd.concat([df_winsorized[['Winsorizada']], forecast_df])

# Graficar la proyección futura
plt.plot(forecast_df.index, forecast_df['Forecast'], label='Pronóstico', color='orange', linestyle='--')

plt.title('Pronóstico de Usuarios Mensuales - Tienda 1 (próximos 12 meses)')
